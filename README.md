### ion.rangeslider
---
https://github.com/IonDen/ion.rangeSlider

```js
$().ionRangeSlider({
  type: "",
  min: 0,
  max: 1000,
  from: 200,
  to: 500,
  grid: true
});
var slider = $("#range").data("ionRangeSlider");
slider.reset();

slider.update({
  from: 300,
  to: 400
});
slider.reset();
slider.destory();

```
